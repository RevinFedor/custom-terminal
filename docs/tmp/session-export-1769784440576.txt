Проблема №1: «Три точки и знак вопроса слева, а должны быть справа»

В текущем коде:

Использование left-1 right-1 растягивает контейнер на всю ширину карточки, а
 justify-end пытается прижать контент вправо.
Почему это ломается: В Grid-верстке или при изменении размеров контейнера 
absolute с привязкой к обоим краям может вести себя непредсказуемо, если у 
родителя (relative) нет явного контекста форматирования блока (BFC) или если
 Flexbox внутри схлопывается. Кроме того, этот слой перекрывает текст 
названия проекта, делая его некликабельным в нижней части.

Решение: Не растягивать контейнер кнопок. Просто прибить его к правому 
нижнему углу.

В Dashboard.tsx:

Паддинг pt-8 (32px) есть, но в Electron-приложениях (особенно на macOS) 
верхняя часть окна часто занята системным "drag-region" (где светофор). Если
 у тебя кастомный тайтлбар, 32px мало — контент "прилипает" к элементам 
управления окном. Также overflow-hidden на родителе может обрезать тени или 
аутлайны.

Решение: Увеличить верхний отступ и убедиться, что скролл-контейнер настроен
 верно.

Проблема №3: «Говнокод» в ProjectCard (Tooltip и Events)

Imperative DOM Manipulation: В блоке onMouseEnter ты используешь 
querySelector, style.right = ..., style.transform = .... Это антипаттерн в 
React. Ты манипулируешь DOM напрямую, обходя React. Это приводит к 
рассинхрону, багам при рендере и трудно поддерживаемому коду.

setTimeout в handleEdit: Использование тайм-аутов для управления состоянием 
UI (закрыть меню -> открыть модалку) — это "костыль", который может привести
 к утечкам памяти или гонкам состояний (race conditions).

Z-Index Wars: Ручное проставление z-index: 50, 51 — признак того, что 
контекст наложения (stacking context) нарушен.

import React, { useState, useRef, useEffect } from 'react';
import { useUIStore } from '../../store/useUIStore';
import { useProjectsStore } from '../../store/useProjectsStore';
import { useWorkspaceStore } from '../../store/useWorkspaceStore';
import { FolderOpen, MoreHorizontal, HelpCircle } from 'lucide-react';

const { ipcRenderer } = window.require('electron');

interface Project {
  id: string;
  path: string;
  name: string;
  description?: string;
}

interface TabsStats {
  total: number;
  active: number;
}

interface ProjectCardProps {
  project: Project;
  isOpen?: boolean;
  onOpen: () => void;
  tabsStats?: TabsStats;
}

export default function ProjectCard({
  project,
  isOpen = false,
  onOpen,
  tabsStats = { total: 0, active: 0 }
}: ProjectCardProps) {
  const [menuOpen, setMenuOpen] = useState(false);
  const [showTooltip, setShowTooltip] = useState(false);
  const [tooltipSide, setTooltipSide] = useState<'left' | 'right'>('left');
  
  const tooltipBtnRef = useRef<HTMLButtonElement>(null);
  const menuRef = useRef<HTMLDivElement>(null);

  const { openEditModal, showToast } = useUIStore();
  const { loadProjects } = useProjectsStore();
  const { openProjects, closeProject } = useWorkspaceStore();

  const hasActiveProcesses = tabsStats.active > 0;

  // Закрытие меню при клике вне его
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (menuRef.current && !menuRef.current.contains(event.target as 
Node)) {
        setMenuOpen(false);
      }
    };
    if (menuOpen) document.addEventListener('mousedown', 
handleClickOutside);
    return () => document.removeEventListener('mousedown', 
handleClickOutside);
  }, [menuOpen]);

  const handleEdit = (e: React.MouseEvent) => {
    e.stopPropagation(); // Останавливаем всплытие, чтобы не открылся проект
    setMenuOpen(false);
    openEditModal(project);
  };

  const handleDelete = async (e: React.MouseEvent) => {
    e.stopPropagation();
    setMenuOpen(false);

    if (!confirm(`Delete "${project.name}"?`)) return;

    if (openProjects.has(project.id)) {
      await closeProject(project.id);
    }

    const result = await ipcRenderer.invoke('project:delete', project.path);
    if (result.success) {
      showToast('Project deleted', 'success');
      loadProjects();
    } else {
      showToast('Failed to delete', 'error');
    }
  };

  // Логика направления тултипа (React Way)
  const handleTooltipEnter = () => {
    if (tooltipBtnRef.current) {
      const rect = tooltipBtnRef.current.getBoundingClientRect();
      const spaceRight = window.innerWidth - rect.right;
      // Если справа места мало (меньше 270px), открываем влево
      setTooltipSide(spaceRight < 270 ? 'left' : 'right');
      setShowTooltip(true);
    }
  };

  const getFolderName = (path: string) => {
    const parts = path.split('/');
    return parts[parts.length - 1] || path;
  };

  // Styles
  const borderColor = isOpen ? 'rgba(168, 85, 247, 0.3)' : 'transparent';
  const bgColor = isOpen ? 'rgba(168, 85, 247, 0.05)' : 
'rgba(255,255,255,0.03)';

  return (
    <div
      className="group relative cursor-pointer transition-all duration-200 
flex flex-col h-[100px]" // Фиксируем высоту или позволяем flex-col
      style={{
        backgroundColor: bgColor,
        border: `1px solid ${borderColor}`,
        borderRadius: '8px',
      }}
      onClick={onOpen}
      onMouseEnter={(e) => {
        if (!isOpen) e.currentTarget.style.borderColor = '#555';
        e.currentTarget.style.backgroundColor = isOpen ? 'rgba(168, 85, 247,
 0.08)' : 'rgba(255,255,255,0.05)';
      }}
      onMouseLeave={(e) => {
        e.currentTarget.style.borderColor = borderColor;
        e.currentTarget.style.backgroundColor = bgColor;
        setShowTooltip(false);
      }}
    >
      {/* Header */}
      <div className="flex items-center gap-2 px-3 pt-3 pb-1">
        <FolderOpen
          size={16}
          className={`flex-shrink-0 transition-colors ${hasActiveProcesses ?
 'text-green-400' : 'text-[#888]'}`}
        />
        <span
          className="text-sm text-white font-medium truncate 
group-hover:text-accent transition-colors w-full"
          title={project.name}
        >
          {project.name}
        </span>
      </div>

      {/* Path */}
      <div className="px-3 text-[10px] text-[#555] truncate mb-auto" 
title={project.path}>
        {getFolderName(project.path)}
      </div>

      {/* Stats - Bottom Left */}
      <div className="px-3 pb-3 mt-1 flex items-center">
        <span className={`text-[10px] ${hasActiveProcesses ? 'text-white' : 
'text-[#666]'}`}>
          {tabsStats.total > 0 ? (
            <>
              {tabsStats.total} tab{tabsStats.total !== 1 ? 's' : ''}
              {hasActiveProcesses && (
                <span className="text-green-400 ml-1">• 
{tabsStats.active}</span>
              )}
            </>
          ) : (
            '—'
          )}
        </span>
      </div>

      {/* Actions - Absolute Bottom Right */}
      {/* ИСПРАВЛЕНИЕ: Фиксируем справа внизу, не растягиваем. Добавил z-10 
чтобы быть выше контента */}
      <div
        className="absolute bottom-2 right-2 flex items-center justify-end 
gap-1 opacity-0 group-hover:opacity-100 transition-opacity z-10"
        onClick={(e) => e.stopPropagation()}
      >
        {/* Menu Button */}
        <div className="relative" ref={menuRef}>
          <button
            className={`p-1.5 rounded transition-all cursor-pointer 
${menuOpen ? 'text-white bg-white/10' : 'text-[#666] hover:text-white 
hover:bg-white/10'}`}
            onClick={(e) => { e.stopPropagation(); setMenuOpen(!menuOpen); 
}}
          >
            <MoreHorizontal size={14} />
          </button>

          {/* Dropdown Menu */}
          {menuOpen && (
            <div className="absolute right-0 bottom-full mb-1 bg-[#1e1e2e] 
border border-[#45475a] rounded-lg shadow-xl min-w-[120px] overflow-hidden 
z-50">
              <button
                className="w-full text-left px-3 py-2 text-xs text-[#ccc] 
hover:bg-white/5 hover:text-white transition-colors border-b border-[#333]"
                onClick={handleEdit}
              >
                Edit
              </button>
              <button
                className="w-full text-left px-3 py-2 text-xs text-[#ff6b6b]
 hover:bg-[#ff6b6b]/10 transition-colors"
                onClick={handleDelete}
              >
                Delete
              </button>
            </div>
          )}
        </div>

        {/* Info Button with React-way Tooltip */}
        {project.description && (
          <div className="relative flex items-center">
            <button
              ref={tooltipBtnRef}
              className="p-1.5 text-[#666] hover:text-white 
hover:bg-white/10 rounded transition-all cursor-pointer"
              onMouseEnter={handleTooltipEnter}
              onMouseLeave={() => setShowTooltip(false)}
            >
              <HelpCircle size={14} />
            </button>

            {/* Tooltip */}
            <div
              className={`absolute bottom-1/2 mb-0 pointer-events-none z-50 
transition-all duration-200 ease-out origin-center
                ${showTooltip ? 'opacity-100 scale-100 translate-y-[-50%]' :
 'opacity-0 scale-95 translate-y-[-40%]'}
              `}
              style={{
                [tooltipSide === 'left' ? 'right' : 'left']: '100%',
                [tooltipSide === 'left' ? 'marginRight' : 'marginLeft']: 
'8px',
                width: 'max-content',
                maxWidth: '260px'
              }}
            >
              <div className="bg-[#252525] border border-[#444] p-3 
rounded-lg shadow-xl backdrop-blur-sm">
                <p className="text-xs text-[#ccc] whitespace-pre-wrap 
leading-relaxed">
                  {project.description}
                </p>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
