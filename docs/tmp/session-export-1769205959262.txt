  // src/renderer/utils/logger.ts
  const LOG_CATEGORIES = {
    claude: false,
    perf: false,
    terminal: false,
    tabs: false,
  } as const;

  type LogCategory = keyof typeof LOG_CATEGORIES;

  // Load from localStorage
  const getEnabled = (): Record<LogCategory, boolean> => {
    try {
      return { ...LOG_CATEGORIES,
  ...JSON.parse(localStorage.getItem('debug') || '{}') };
    } catch {
      return { ...LOG_CATEGORIES };
    }
  };

  let enabled = getEnabled();

  export const log = {
    claude: (...args: any[]) => enabled.claude && console.log('[Claude]',
  ...args),
    perf: (...args: any[]) => enabled.perf && console.log('[Perf]',
  ...args),
    terminal: (...args: any[]) => enabled.terminal &&
  console.log('[Terminal]', ...args),
    tabs: (...args: any[]) => enabled.tabs && console.log('[Tabs]',
  ...args),

    // Toggle in console: log.enable('claude', 'perf')
    enable: (...categories: LogCategory[]) => {
      categories.forEach(c => enabled[c] = true);
      localStorage.setItem('debug', JSON.stringify(enabled));
    },
    disable: (...categories: LogCategory[]) => {
      categories.forEach(c => enabled[c] = false);
      localStorage.setItem('debug', JSON.stringify(enabled));
    },
    // Show current state
    status: () => console.table(enabled),
  };

  // Expose globally for console access
  (window as any).log = log;

  Использование в коде:
  import { log } from '../utils/logger';

  log.perf('Switch took:', performance.now() - start, 'ms');

  log.enable('claude', 'perf')  // включить
  log.disable('perf')           // выключить
  log.status()                  // показать что включено

  Включение/выключение

  В терминале (main process):
  DEBUG=app:claude npm run dev           # только claude
  DEBUG=app:* npm run dev                # все app:*
  DEBUG=app:claude,app:perf npm run dev  # несколько
  DEBUG=* npm run dev                    # вообще всё

  В браузере/Electron renderer (DevTools консоль):
  localStorage.debug = 'app:claude'       // включить
  localStorage.debug = 'app:*'            // все
  localStorage.debug = ''                 // выключить
  // Потом F5 (перезагрузка страницы)

  - Показывает время между вызовами (+15ms)
