  magenta: '\x1b[35m'

const log = {
  step: (m) => console.log(`${c.cyan}[STEP]${c.reset} ${m}`),
  info: (m) => console.log(`${c.dim}[INFO]${c.reset} ${m}`),
  pass: (m) => console.log(`${c.green}[PASS]${c.reset} ${m}`),
  fail: (m) => console.log(`${c.red}[FAIL]${c.reset} ${m}`),
  warn: (m) => console.log(`${c.yellow}[WARN]${c.reset} ${m}`),
  data: (m) => console.log(`${c.magenta}[DATA]${c.reset} ${m}`)
}

// Read visible xterm content
async function readTerminalContent(page) {
  return page.evaluate(() => {
    const rows = document.querySelectorAll('.xterm-rows > div')
    const lines = []
    rows.forEach(row => {
      const text = row.textContent
      if (text?.trim()) lines.push(text)
    })
    return lines.slice(-30).join('\n')
  })
}

// Send raw key to PTY via IPC
async function sendKeyToPty(page, tabId, key) {
  await page.evaluate(({ tid, k }) => {
    const { ipcRenderer } = window.require('electron')
    ipcRenderer.send('terminal:input', tid, k)
  }, { tid: tabId, k: key })
}

// Call claude:open-history-menu IPC and return result
async function callRewindIpc(page, tabId, targetIndex, targetText, pasteAfter) {
  return page.evaluate(({ tid, ti, tt, pa }) => {
    const { ipcRenderer } = window.require('electron')
    return ipcRenderer.invoke('claude:open-history-menu', {
      tabId: tid,
      targetIndex: ti,
      targetText: tt,
      pasteAfter: pa
    })
  }, { tid: tabId, ti: targetIndex, tt: targetText, pa: pasteAfter })
}

// Get timeline entries from store
async function getTimelineEntries(page) {
  return page.evaluate(() => {
    const store = window.useWorkspaceStore?.getState?.()
    if (!store) return null
    const proj = store.openProjects?.get?.(store.activeProjectId)
    if (!proj) return null
    const tab = proj.tabs?.get?.(proj.activeTabId)
    if (!tab) return null
    return {
      sessionId: tab.claudeSessionId,
      commandType: tab.commandType,
      tabId: proj.activeTabId
    }
  })
}

// Wait for Claude to finish responding (check for prompt symbol in terminal)
async function waitForPrompt(page, timeout = 30000) {
  const start = Date.now()
  while (Date.now() - start < timeout) {
    const content = await readTerminalContent(page)
    // Claude prompt symbols: ⌵ (idle), ⏵ (thinking), or the Unicode cursor
    if (content.includes('\u2335') || content.includes('\u23F5') || content.includes('╰')) {
      // Check the last few lines for a prompt indicator
      const lastLines = content.split('\n').slice(-5).join('\n')

function getSearchText(content: string): string {
  const lines = content.split('\n');
  for (const line of lines) {
    const t = line.trim();
    if (!t || t.length < 3) continue;
